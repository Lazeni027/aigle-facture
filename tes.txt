    // ******************* AIGLE SERVICE - PROFORMA & PAIEMENT ********************
    let items = [];       // chaque élément: { description, quantity, unitPrice }
    let currentDocType = "proforma";   // proforma ou payment
    
    const companyInfo = {

    // DOM references
    const itemsBody = document.getElementById('itemsBody');
    const totalHTSpan = document.getElementById('totalHT');
    const totalVATSpan = document.getElementById('totalVAT');
    const totalDiscountSpan = document.getElementById('totalDiscount');
    const grandTotalSpan = document.getElementById('grandTotalSpan');
    const taxRateInput = document.getElementById('taxRate');
    const discountAmountInput = document.getElementById('discountAmountInput');
    const taxPercentSpan = document.getElementById('taxPercentSpan');
    const docNumberInput = document.getElementById('docNumber');
    const docDateInput = document.getElementById('docDate');
    const clientNameInput = document.getElementById('clientName');
    const clientEmailInput = document.getElementById('clientEmail');
    const clientAddressInput = document.getElementById('clientAddress');
    const notesField = document.getElementById('notesField');
    const currentDocTypeLabel = document.getElementById('currentDocTypeLabel');
    const livePreviewDiv = document.getElementById('livePreview');

    function setDefaultDate() {
        if (!docDateInput.value) {
            const today = new Date().toISOString().slice(0,10);
            docDateInput.value = today;
        }
    }
    
    function computeTotals() {
        let totalHT = 0;
        items.forEach(item => {
            const qty = parseFloat(item.quantity) || 0;
            const price = parseFloat(item.unitPrice) || 0;
            totalHT += qty * price;
        });
        const taxRate = parseFloat(taxRateInput.value) || 0;
        const discountAmount = parseFloat(discountAmountInput.value) || 0;
        
        // Appliquer remise en montant (soustraction directe)
        let afterDiscountHT = totalHT - discountAmount;
        if (afterDiscountHT < 0) afterDiscountHT = 0;
        
        const vatAmount = afterDiscountHT * (taxRate / 100);
        const totalTTC = afterDiscountHT + vatAmount;
        
        totalHTSpan.innerText = totalHT.toLocaleString('fr-FR', {minimumFractionDigits:2, maximumFractionDigits:2}) + " FCFA";
        totalVATSpan.innerText = vatAmount.toLocaleString('fr-FR', {minimumFractionDigits:2}) + " FCFA";
        totalDiscountSpan.innerText = discountAmount.toLocaleString('fr-FR', {minimumFractionDigits:2}) + " FCFA";
        grandTotalSpan.innerText = totalTTC.toLocaleString('fr-FR', {minimumFractionDigits:2}) + " FCFA";
        taxPercentSpan.innerText = taxRate;
        
        return { totalHT, afterDiscountHT, vatAmount, totalTTC, discountAmount, taxRate };
    }
    
    function escapeHtml(str) { 
        if(!str) return '';
        return str.replace(/[&<>]/g, function(m){
            if(m==='&') return '&amp;';
            if(m==='<') return '&lt;';
            if(m==='>') return '&gt;';
            return m;
        }); 
    }
    
    function renderEditableTable() {
        itemsBody.innerHTML = '';
        if (!items.length) {
            const emptyRow = document.createElement('tr');
            emptyRow.innerHTML = `<td colspan="5" style="text-align:center; padding:18px; color:#7c9b8e;">➕ Ajoutez une prestation ci-dessus</td>`;
            itemsBody.appendChild(emptyRow);
        } else {
            items.forEach((item, idx) => {
                const row = document.createElement('tr');
                const descInput = document.createElement('input');
                descInput.type = 'text';
                descInput.className = 'item-desc-input';
                descInput.value = item.description || '';
                descInput.placeholder = 'ex: Prestation AIGLE SERVICE';
                descInput.addEventListener('input', (e) => {
                    items[idx].description = e.target.value;
                    updatePreview();
                    computeTotals();
                });
                const descCell = document.createElement('td');
                descCell.appendChild(descInput);
                
                const qtyInput = document.createElement('input');
                qtyInput.type = 'number';
                qtyInput.step = '1';
                qtyInput.min = '0';
                qtyInput.className = 'item-qty-input';
                qtyInput.value = item.quantity;
                qtyInput.addEventListener('input', (e) => {
                    let val = parseFloat(e.target.value);
                    if (isNaN(val)) val = 0;
                    items[idx].quantity = val;
                    computeTotals();
                    updatePreview();
                    updateRowTotalDisplay(row, idx);
                });
                const qtyCell = document.createElement('td');
                qtyCell.appendChild(qtyInput);
                
                const priceInput = document.createElement('input');
                priceInput.type = 'number';
                priceInput.step = '10';
                priceInput.min = '0';
                priceInput.className = 'item-price-input';
                priceInput.value = item.unitPrice;
                priceInput.addEventListener('input', (e) => {
                    let val = parseFloat(e.target.value);
                    if (isNaN(val)) val = 0;
                    items[idx].unitPrice = val;
                    computeTotals();
                    updatePreview();
                    updateRowTotalDisplay(row, idx);
                });
                const priceCell = document.createElement('td');
                priceCell.appendChild(priceInput);
                
                const totalCell = document.createElement('td');
                totalCell.className = 'right';
                totalCell.style.fontWeight = '500';
                const qtyVal = parseFloat(item.quantity) || 0;
                const priceVal = parseFloat(item.unitPrice) || 0;
                totalCell.innerText = (qtyVal * priceVal).toLocaleString('fr-FR', {minimumFractionDigits:0}) + ' FCFA';
                
                const deleteCell = document.createElement('td');
                deleteCell.className = 'item-delete';
                const trashIcon = document.createElement('i');
                trashIcon.className = 'fas fa-trash-alt';
                trashIcon.style.cursor = 'pointer';
                trashIcon.addEventListener('click', () => {
                    items.splice(idx, 1);
                    refreshAll();
                });
                deleteCell.appendChild(trashIcon);
                
                row.appendChild(descCell);
                row.appendChild(qtyCell);
                row.appendChild(priceCell);
                row.appendChild(totalCell);
                row.appendChild(deleteCell);
                itemsBody.appendChild(row);
            });
        }
        attachDynamicRowUpdates();
    }
    
    function updateRowTotalDisplay(row, idx) {
        if (row && idx < items.length) {
            const totalCell = row.cells[3];
            if (totalCell) {
                const qty = parseFloat(items[idx].quantity) || 0;
                const price = parseFloat(items[idx].unitPrice) || 0;
                totalCell.innerText = (qty * price).toLocaleString('fr-FR', {minimumFractionDigits:0}) + ' FCFA';
            }
        }
    }
    
    function attachDynamicRowUpdates() {
        const rows = itemsBody.querySelectorAll('tr');
        rows.forEach((row, idx) => {
            if (idx >= items.length) return;
            const qty = parseFloat(items[idx].quantity) || 0;
            const price = parseFloat(items[idx].unitPrice) || 0;
            const totalCell = row.cells[3];
            if (totalCell) totalCell.innerText = (qty * price).toLocaleString('fr-FR', {minimumFractionDigits:0}) + ' FCFA';
        });
    }
    
    function getDocumentTitleAndSub() {
        if (currentDocType === "proforma") {
            return { 
                title: "FACTURE PROFORMA", 
                sub: "Document proforma – En attente de règlement", 
                badge: "PROFORMA",
                paymentSpecific: false
            };
        } else {
            return { 
                title: "REÇU DE PAIEMENT", 
                sub: "Document attestant du paiement – Quittance valable", 
                badge: "PAIEMENT",
                paymentSpecific: true
            };
        }
    }
    
    function updatePreview() {
        const totals = computeTotals();
        const docNumber = docNumberInput.value.trim() || "AIGLE-001";
        const docDate = docDateInput.value || "Date non définie";
        const client = clientNameInput.value.trim() || "Client non spécifié";
        const clientEmail = clientEmailInput.value.trim() || "";
        const clientAddr = clientAddressInput.value.trim() || "";
        const notes = notesField.value.trim() || "";
        const taxRate = totals.taxRate;
        const discountAmount = totals.discountAmount;
        const totalHT = totals.totalHT;
        const afterDiscountHT = totals.afterDiscountHT;
        const vatAmount = totals.vatAmount;
        const totalTTC = totals.totalTTC;
        const { title, sub, paymentSpecific } = getDocumentTitleAndSub();
        
        let rowsHtml = "";
        items.forEach(item => {
            const qty = parseFloat(item.quantity) || 0;
            const price = parseFloat(item.unitPrice) || 0;
            const totalLine = (qty * price).toFixed(2);
            const descriptionDisplay = item.description && item.description.trim() !== "" ? item.description : "—";
            rowsHtml += `<tr><td>${escapeHtml(descriptionDisplay)}</td><td class="right">${qty}</td><td class="right">${price.toFixed(2)} FCFA</td><td class="right">${parseFloat(totalLine).toLocaleString('fr-FR')} FCFA</td></tr>`;
        });
        if(items.length === 0) rowsHtml = "<tr><td colspan='4' style='text-align:center;padding:20px;'>Aucun article – ajoutez une prestation</td></tr>";
        
        let paymentExtraHtml = "";
        if (paymentSpecific) {
            paymentExtraHtml = `
                <div style="background: #e8f5e9; padding: 14px; border-radius: 16px; margin: 15px 0; text-align: center; border: 1px solid #a5d6a7;">
                    <i class="fas fa-check-circle" style="color:#2e7d32; font-size: 1.3rem;"></i>
                    <strong style="color:#1b5e20;"> PAIEMENT ACCUSÉ</strong><br>
                    <span style="font-size:0.85rem;">Montant réglé : ${totalTTC.toLocaleString('fr-FR', {minimumFractionDigits:2})} FCFA</span><br>
                    <span style="font-size:0.7rem;">Date de règlement : ${docDate} | Mode: Mobile Money / Espèces</span>
                </div>
            `;
        } else {
            paymentExtraHtml = `<div class="payment-highlight" style="background:#fef5e8; margin-top:12px;"><i class="fas fa-hourglass-half"></i> Document proforma – En attente de validation du paiement</div>`;

    function generatePDF() {
        const previewElem = document.getElementById('livePreview');
        if (!previewElem) return;
        const opt = {
            margin:        [0.5, 0.5, 0.5, 0.5],
            filename:     `${currentDocType === "proforma" ? "PROFORMA" : "PAIEMENT"}_${docNumberInput.value.trim() || 'document'}_AIGLE.pdf`,
            image:        { type: 'jpeg', quality: 0.98 },
            html2canvas:  { scale: 2, letterRendering: true },
            jsPDF:        { unit: 'in', format: 'a4', orientation: 'portrait' }
        };
        html2pdf().set(opt).from(previewElem).save();
    }
    
    function addNewEmptyLine() {
        items.push({ description: "", quantity: 1, unitPrice: 0 });
        refreshAll();
        setTimeout(() => {
            const lastDesc = document.querySelector('#itemsBody tr:last-child .item-desc-input');
            if (lastDesc) lastDesc.focus();
        }, 60);
    }
    
    function setDocType(type) {
        currentDocType = type;
        let prefix = "AIGLE-";
        let oldVal = docNumberInput.value;
        if (!oldVal.startsWith(prefix)) {
            let clean = oldVal.replace(/^(AIGLE-)/, '');
            docNumberInput.value = prefix + (clean || "001");
        }
        updatePreview();
    }
    
    function resetForm() {
        items = [{ description: "", quantity: 1, unitPrice: 0 }];
        docNumberInput.value = "AIGLE-001";
        setDefaultDate();
        clientNameInput.value = "";
        clientEmailInput.value = "";
        clientAddressInput.value = "";
        notesField.value = "Paiement : Orange Money, Wave. Merci.";
        taxRateInput.value = "0";
        discountAmountInput.value = "0";
        currentDocType = "proforma";
        refreshAll();
        setDocType("proforma");
    }
